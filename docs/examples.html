Fibonacci
func fib(n) {
  if n <= 1 { return n }
  return fib(n-1) + fib(n-2)
}
for i in range(0, 10) { print(fib(i)) }

List Operations

print("Avg: " + total / scores.len())
Random Number Game
let secret = random(1, 100)
let guess = 0
while guess != secret {
  guess = input("Guess 1-100: ")
  if guess < secret { print("Too low!") }
  else if guess > secret { print("Too high!") }
}
print("Correct! 🎉")
Countdown
let n = 10
while n >= 0 {
  if n == 0 { print("LAUNCH! 🚀") }
  else { print(n) }
  n = n - 1
}
